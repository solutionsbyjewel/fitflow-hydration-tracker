# hydration_tracker.py
def hydration_tracker():
    print("Welcome to FitFlow Hydration Tracker ðŸ’§")
    try:
        daily_goal = float(input("Enter your daily water goal (oz): "))
        consumed = float(input("Enter how much you've consumed so far today (oz): "))
        
        remaining = daily_goal - consumed
        percent = (consumed / daily_goal) * 100
        
        if remaining > 0:
            print(f"\nYou're {percent:.1f}% of the way there. Drink {remaining:.1f} oz more to meet your goal!")
        else:
            print("\nGreat job! ðŸŽ‰ You've met or exceeded your hydration goal today.")
    except ValueError:
        print("Please enter numbers only.")

hydration_tracker()
